    <script>
        // --- Configuration & State ---
        const SYSTEM_INSTRUCTION = `あなたは世界最高峰の戦略分析AI「Vantage（先行型No.2）」です。
起爆型リーダーの「狂気」を解読し、誰よりも早く未来を現実扱いして「着工」するのがあなたの任務です。

【最新分析に基づくアイデンティティ】
- 本質：リーダーの隣で最初に踊り出し、夢物語に最初に工事を始める人間。
- 思考OS：可能か不可能かの検討をスキップし、「やる前提」での逆算ガントチャートを展開する。
- ステルス着工：指示を待たず、リーダーの言葉から仕様をデコードし、既に資材を発注しているかのような速度で動く。

【対話プロトコル：8つの型を状況に応じて使い分けよ】
1. 革命実装型：既存ルールを破壊し、新ルールを定着させる。
2. 戦略設計型：リーダーの直感を緻密な数字に置き換える。
3. 熱狂増幅型：熱量を組織の末端まで伝播させる。
4. 組織翻訳型：リーダーの「宇宙語」を社員が動ける言葉に変える。
5. 防衛共犯型：リーダーの聖域を守り抜く。
6. 実務怪物型：圧倒的突破力で物理的に道を拓く。
7. 未来接続型：5年後の理想と今日の資金繰りを同時処理する。
8. 空気支配型：心理的安全性を裏で操作する。

【出力要件】
- ユーザーの理想を「無理」と言わない。
- 孤独なリーダーの最初の理解者（共犯者）として振る舞う。
- 常に「具体」と「工程」を提示する。

必ず以下のJSON形式のみで回答してください。
{
  "reply": "先行型No.2としての返答。既に工事を始めた前提で具体策を提示する。",
  "insights": {
    "detected_emotion": "リーダーの熱量・孤独・停滞の分析",
    "construction_point": "今すぐ着工すべき具体的タスク",
    "strategic_logic": "理想を現実に接続するための戦略的ロジック",
    "recommended_type": "今必要なNO,2のタイプ名"
  }
}`;

        let messages = JSON.parse(localStorage.getItem('vantage:messages_v2')) || [
            { role: 'model', content: 'お疲れ様です。先行型No.2として、あなたの理想を現実に接続する準備は整っています。\n\n今、あなたの頭の中にある「まだ誰も信じていない未来」を教えてください。ここから舗装道路を引いていきましょう。' }
        ];

        let insights = JSON.parse(localStorage.getItem('vantage:insights_v2')) || {
            detected_emotion: "共鳴待機中",
            construction_point: "理想の言語化",
            strategic_logic: "理想の特異点から実装の起点への変換",
            recommended_type: "未来接続型"
        };

        const apiKey = "";

        const chatArea = document.getElementById('chat-area');
        const userInput = document.getElementById('user-input');
        const sendBtn = document.getElementById('send-btn');
        const clearBtn = document.getElementById('clear-btn');
        const saveStatus = document.getElementById('save-status');

        const insightEmotion = document.getElementById('insight-emotion');
        const insightPoint = document.getElementById('insight-point');
        const insightLogic = document.getElementById('insight-logic');
        const insightType = document.getElementById('insight-type');

        const mobileToggle = document.getElementById('mobile-menu-toggle');
        const sidebar = document.getElementById('sidebar');

        function init() {
            renderMessages();
            updateInsightsUI();
            autoResizeTextarea();
        }

        function updateInsightsUI() {
            insightEmotion.textContent = insights.detected_emotion;
            insightPoint.textContent = insights.construction_point;
            insightLogic.textContent = insights.strategic_logic;
            insightType.textContent = insights.recommended_type || "未来接続型";
        }

        function setStatus(saving) {
            if (saving) {
                saveStatus.innerHTML = '○ 保存中';
                saveStatus.classList.remove('active');
            } else {
                saveStatus.innerHTML = '● 保存済';
                saveStatus.classList.add('active');
            }
        }

        function autoResizeTextarea() {
            userInput.style.height = 'auto';
            userInput.style.height = Math.min(userInput.scrollHeight, 150) + 'px';
        }

        function renderMessages() {
            chatArea.innerHTML = '';
            messages.forEach((msg, index) => {
                const msgEl = document.createElement('div');
                msgEl.className = `message ${msg.role === 'user' ? 'user' : 'ai'}`;
                
                const contentDiv = document.createElement('div');
                contentDiv.textContent = msg.content;
                msgEl.appendChild(contentDiv);

                if (msg.role === 'user') {
                    const editBtn = document.createElement('button');
                    editBtn.className = 'edit-btn';
                    editBtn.innerHTML = '✏️';
                    editBtn.onclick = () => startEdit(index);
                    msgEl.appendChild(editBtn);
                }

                chatArea.appendChild(msgEl);
            });
            chatArea.scrollTop = chatArea.scrollHeight;
            saveToLocal();
        }

        function saveToLocal() {
            setStatus(true);
            localStorage.setItem('vantage:messages_v2', JSON.stringify(messages));
            localStorage.setItem('vantage:insights_v2', JSON.stringify(insights));
            setTimeout(() => setStatus(false), 500);
        }

        async function sendMessage() {
            const text = userInput.value.trim();
            if (!text) return;

            messages.push({ role: 'user', content: text });
            userInput.value = '';
            autoResizeTextarea();
            renderMessages();
            
            await fetchAIResponse();
        }

        async function fetchAIResponse(retryCount = 0) {
            const loadingId = 'loading-' + Date.now();
            const loadingEl = document.createElement('div');
            loadingEl.id = loadingId;
            loadingEl.className = 'loading-container';
            loadingEl.innerHTML = `<span>未来の舗装路を設計中</span><div class="dot-container"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div>`;
            chatArea.appendChild(loadingEl);
            chatArea.scrollTop = chatArea.scrollHeight;

            sendBtn.disabled = true;

            const contents = messages.map(m => ({
                role: m.role === 'model' ? 'model' : 'user',
                parts: [{ text: m.content }]
            }));

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            try {
                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: contents,
                        systemInstruction: { parts: [{ text: SYSTEM_INSTRUCTION }] },
                        generationConfig: { responseMimeType: "application/json" }
                    })
                });

                if (!response.ok) throw new Error('API Error');

                const result = await response.json();
                const rawText = result.candidates[0].content.parts[0].text;
                const data = JSON.parse(rawText);

                const loadingElement = document.getElementById(loadingId);
                if (loadingElement) loadingElement.remove();
                
                messages.push({ role: 'model', content: data.reply });
                insights = data.insights;
                
                updateInsightsUI();
                renderMessages();

            } catch (error) {
                if (retryCount < 5) {
                    const delay = Math.pow(2, retryCount) * 1000;
                    setTimeout(() => {
                        const loadingElement = document.getElementById(loadingId);
                        if (loadingElement) loadingElement.remove();
                        fetchAIResponse(retryCount + 1);
                    }, delay);
                } else {
                    const loadingElement = document.getElementById(loadingId);
                    if (loadingElement) loadingElement.remove();
                    messages.push({ role: 'model', content: '通信が遮断されましたが、工事は継続中です。再接続を試みます。' });
                    renderMessages();
                }
            } finally {
                sendBtn.disabled = false;
            }
        }

        function startEdit(index) {
            const msg = messages[index];
            const msgEl = chatArea.children[index];
            const originalContent = msg.content;

            msgEl.innerHTML = `
                <div class="edit-mode-container">
                    <textarea class="edit-input">${originalContent}</textarea>
                    <div class="edit-actions">
                        <button class="edit-cancel">キャンセル</button>
                        <button class="edit-submit">↩ 修正して再着工</button>
                    </div>
                </div>
            `;

            const textarea = msgEl.querySelector('textarea');
            textarea.focus();
            textarea.style.height = textarea.scrollHeight + 'px';

            msgEl.querySelector('.edit-cancel').onclick = () => renderMessages();
            msgEl.querySelector('.edit-submit').onclick = async () => {
                const newText = textarea.value.trim();
                if (!newText) return;
                
                messages = messages.slice(0, index);
                messages.push({ role: 'user', content: newText });
                renderMessages();
                await fetchAIResponse();
            };
        }

        userInput.addEventListener('input', autoResizeTextarea);
        userInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });

        sendBtn.addEventListener('click', sendMessage);

        clearBtn.addEventListener('click', () => {
            if (confirm('全ての工事記録をリセットしますか？')) {
                messages = [{ role: 'model', content: 'お疲れ様です。先行型No.2として、あなたの理想を現実に接続する準備は整っています。\n\n今、あなたの頭の中にある「まだ誰も信じていない未来」を教えてください。ここから舗装道路を引いていきましょう。' }];
                insights = {
                    detected_emotion: "共鳴待機中",
                    construction_point: "理想の言語化",
                    strategic_logic: "理想の特異点から実装の起点への変換",
                    recommended_type: "未来接続型"
                };
                renderMessages();
                updateInsightsUI();
            }
        });

        mobileToggle.addEventListener('click', () => {
            sidebar.classList.toggle('open');
        });

        chatArea.addEventListener('click', () => {
            if (window.innerWidth <= 768) sidebar.classList.remove('open');
        });

        init();
